Győr Switch konfigurációk

Győr_RG_Switch
en
clock set 12:00:00 25 Jan 2026
conf t
hostname Gyor_RG_Switch
no ip domain-lookup
enable secret Cisco123
service password-encryption
banner motd #Belepes csak engedellyel!#

line console 0
password Class123
login
exit

line vty 0 15
password Class123
login local
transport input ssh
exit

ip domain-name corelink-kft.hu
crypto key generate rsa
1024

user admin priv 15 secret Admin123

ip ssh version 2

int vlan1
ip add 172.16.1.2 255.255.255.224
no sh
exit

ip default-gateway 172.16.1.1
